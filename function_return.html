JavaScript return 문의 기능과 반환 형태

1. return의 기본 기능

return 문은 함수의 실행을 종료하고, 지정된 값을 함수의 호출 지점으로 반환합니다.

function greet(name) {
    return "안녕하세요, " + name + "님!";
}

2. 다양한 반환 형태

2.1 문자열 반환

function returnString() {
    return "이것은 문자열입니다.";
}

2.2 숫자 반환

function returnNumber() {
    return 42;
}

2.3 불리언 반환

function returnBoolean(value) {
    return value > 10;
}

2.4 객체 반환

function returnObject() {
    return { name: "홍길동", age: 30 };
}

2.5 배열 반환

function returnArray() {
    return [1, 2, 3, 4, 5];
}

2.6 함수 반환 (고차 함수)

function returnFunction() {
    return function(x) { return x * x; };
}

2.7 undefined 반환

function returnUndefined() {
    return;
}
